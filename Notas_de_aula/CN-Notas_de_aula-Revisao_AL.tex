\subsection*{Inversão de Matrizes}

Resolução de um sistema linear usando a matriz inversa

\begin{displaymath}
  Ax=b
\end{displaymath}
\begin{displaymath}
  A^{-1}Ax=A^{-1}b
\end{displaymath}
\begin{displaymath}
  x=A^{-1}b
\end{displaymath}


\begin{displaymath}
  A=\begin{bmatrix}
  2&3\\
  3&4\\
  \end{bmatrix}
\end{displaymath}

\begin{displaymath}
  [A|I]=\begin{bmatrix}
  2&3&|&1&0\\
  3&4&|&0&1\\
  \end{bmatrix}
\end{displaymath}

\begin{displaymath}
  \begin{bmatrix}
    1&3/2&|&1/2&0\\
    3&4&|&0&1\\
  \end{bmatrix}
\end{displaymath}

\begin{displaymath}
  \begin{bmatrix}
    1&3/2&|&1/2&0\\
    0&1/2&|&3/2&-1\\
  \end{bmatrix}
\end{displaymath}

\begin{displaymath}
  \begin{bmatrix}
    1&3/2&|&1/2&0\\
    0&1&|&3&-2\\
  \end{bmatrix}
\end{displaymath}

\begin{displaymath}
  \begin{bmatrix}
    1&0&|&-4&3\\
    0&1&|&3&-2\\
  \end{bmatrix}
\end{displaymath}


\begin{displaymath}
  A^{-1}=
  \begin{bmatrix}
    -4&3\\
    3&-2\\
  \end{bmatrix}
\end{displaymath}
